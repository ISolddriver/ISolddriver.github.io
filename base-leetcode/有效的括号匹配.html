# 有效的括号匹配

给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串 s ，判断字符串是否有效。 有效字符串需满足：

  1. 左括号必须用相同类型的右括号闭合。
  2. 左括号必须以正确的顺序闭合。
  3. 每个右括号都有一个对应的相同类型的左括号。

示例 1：
输入：s = "([])"
输出：true

思路：使用栈，遇到左括号入栈，遇到右括号出栈，最后栈为空则说明是有效的括号
必须是偶数，否则直接返回false

function isValid(s) {
  if (s.length % 2 !== 0) return false;
  const stack = [];
  const map = {
    ')': '(',
    '}': '{',
    ']': '['
  }
  for(let i = 0; i < s.length; i++) {
    if (map[s[i]]) {
      if (map[s[i]] !== stack[stack.length - 1]) {
        return false
      } else {
        stack.pop()
      }
    } else {
      stack.push(s[i])
    }
  }
  return stack.length === 0
}
